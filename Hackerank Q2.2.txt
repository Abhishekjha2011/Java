import java.io.*;

import java.util.*;

public class Solution {

    public static void main(String[] args) {

        

        Scanner sc=new Scanner(System.in);

        int a=sc.nextInt();

        if(a<0)

        {

            System.out.println(a*(-1));

        }

        else{

            System.out.println(a);

        }

    }
